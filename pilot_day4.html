import matplotlib.pyplot as plt

def plot_cmd(photo_table):
    y = photo_table['g_mean_psf_mag']
    x = photo_table['g_mean_psf_mag'] - photo_table['i_mean_psf_mag']

    plt.plot(x, y, 'ko', markersize=0.3, alpha=0.3)

    plt.xlim(0, 1.5)
    plt.ylim(14, 22)

    plt.gca().invert_yaxis()

    plt.ylabel(r'$g_0$')
    plt.xlabel(r'$(g-i)_0$');
